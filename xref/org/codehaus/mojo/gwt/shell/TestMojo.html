395     @Override
396     public void doExecute()
397         throws MojoExecutionException, MojoFailureException
398     {
399         if ( skip || skipTests || skipExec )
400         {
401             return;
402         }
403         new TestTemplate( getProject(), includes, excludes, new TestTemplate.CallBack()
404         {
405             public void doWithTest( File sourceDir, String test )
406                 throws MojoExecutionException
407             {
408                 forkToRunTest( test );
409             }
410         } );
411 
412         if ( failures > 0 )
413         {
414             if ( testFailureIgnore )
415             {
416                 getLog().error( "There are test failures.\n\nPlease refer to " + reportsDirectory
417                                     + " for the individual test results." );
418             }
419             else
420             {
421                 throw new MojoExecutionException( "There was test failures." );
422             }
423         }
424     }

426     /**
427      * @param classpath the test execution classpath
428      * @param jvm the JVM process command
429      * @param test the test to run
430      * @throws MojoExecutionException some error occured
431      */
432     private void forkToRunTest( String test )
433         throws MojoExecutionException
434     {
435         test = test.substring( 0, test.length() - 5 );
436         test = StringUtils.replace( test, File.separator, "." );
437         try
438         {
439             File outFile = new File(out);
440             if (outFile.isAbsolute())
441             {
442                 outFile.mkdirs();
443             }
444             else
445             {
446                 new File( getProject().getBasedir(), out ).mkdirs();
447             }
448             try
449             {
450                 JavaCommand cmd = createJavaCommand()
451                     .setMainClass( MavenTestRunner.class.getName() );
452                 if ( gwtSdkFirstInClasspath )
453                 {
454                     cmd.addToClasspath( getGwtUserJar() )
455                        .addToClasspath( getGwtDevJar() );
456                 }
457                 cmd.addToClasspath( getClasspath( Artifact.SCOPE_TEST ) );
458                 if ( !gwtSdkFirstInClasspath )
459                 {
460                     cmd.addToClasspath( getGwtUserJar() )
461                        .addToClasspath( getGwtDevJar() );
462                 }
463 
464                 addCompileSourceArtifacts( cmd );
465 
466                 cmd.arg( test );
467                 cmd.systemProperty( "surefire.reports", reportsDirectory.getAbsolutePath() );
468                 cmd.systemProperty( "gwt.args", getGwtArgs() );
469 
470                 cmd.execute();
471             }
472             catch ( ForkedProcessExecutionException e )
473             {
474                 failures++;
475             }
476         }
477         catch ( Exception e )
478         {
479             throw new MojoExecutionException( "Failed to run GWT tests", e );
480         }
481     }

483     protected String getGwtArgs()
484     {
485         StringBuilder sb = new StringBuilder();
486         sb.append( "-war " ).append( quote( out ) );
487         sb.append( " -logLevel " ).append( quote( getLogLevel() ) );
488         sb.append( ( webMode || productionMode ) ? " -nodevMode" : " -devMode" );
489         sb.append( checkAssertions ? " -checkAssertions" : " -nocheckAssertions" );
490         sb.append( clusterFunctions ? " -XclusterFunctions" : " -XnoclusterFunctions" );
491         sb.append( disableCastChecking ? " -XnocheckCasts" : " -XcheckCasts" );
492         sb.append( disableClassMetadata ? " -XnoclassMetadata" : " -XclassMetadata" );
493         sb.append( disableRunAsync ? " -XnocodeSplitting" : " -XcodeSplitting" );
494         sb.append( draftCompile ? " -draftCompile" : " -nodraftCompile" );
495         sb.append( inlineLiteralParameters ? " -XinlineLiteralParameters" : " -XnoinlineLiteralParameters" );
496         sb.append( optimizeDataflow ? " -XoptimizeDataflow" : " -XnooptimizeDataflow" );
497         sb.append( ordinalizeEnums ? " -XordinalizeEnums" : " -XnoordinalizeEnums" );
498         sb.append( removeDuplicateFunctions ? " -XremoveDuplicateFunctions" : " -XnoremoveDuplicateFunctions" );
499         sb.append( showUi ? " -showUi" : " -noshowUi" );
500         sb.append( " -sourceLevel " ).append( quote( sourceLevel ) );
501         sb.append( " -testBeginTimeout " ).append( testBeginTimeout );
502         sb.append( " -testMethodTimeout ").append( testMethodTimeout );
503         sb.append( " -Xtries " ).append( tries );
504         sb.append( incremental ? " -incremental" : " -noincremental" );
505         sb.append( generateJsInteropExports ? " -generateJsInteropExports" : " -nogenerateJsInteropExports" );
506 
507         if ( optimizationLevel >= 0 )
508         {
509             sb.append( " -optimize " ).append( optimizationLevel );
510         }
511         if ( precompile != null && !precompile.trim().isEmpty() )
512         {
513             sb.append( " -precompile " ).append( quote( precompile ) );
514         }
515         if ( logDir != null )
516         {
517             sb.append( " -logdir " ).append( quote( logDir.getAbsolutePath() ) );
518         }
519         if ( workDir != null )
520         {
521             sb.append( " -workDir " ).append( quote( workDir.getAbsolutePath() ) );
522         }
523 
524         if ( namespace != null && !namespace.trim().isEmpty() )
525         {
526             sb.append( " -Xnamespace " ).append( quote( namespace ) );
527         }
528         if ( jsInteropMode != null && jsInteropMode.length() > 0 && !jsInteropMode.equals( "JS_RC" ) )
529         {
530             sb.append( " -XjsInteropMode " ).append( quote( jsInteropMode ) );
531         }
532 
533         if ( mode.equalsIgnoreCase( "manual" ) )
534         {
535             sb.append( " -runStyle Manual:1" );
536         }
537         else if ( mode.equalsIgnoreCase( "htmlunit" ) )
538         {
539             sb.append( " -runStyle ").append( quote( "HtmlUnit:" + htmlunit ) );
540         }
541         else if ( mode.equalsIgnoreCase( "selenium" ) )
542         {
543             sb.append( " -runStyle ").append( quote( "Selenium:" + selenium ) );
544         }
545         else if ( !mode.trim().isEmpty() )
546         {
547             sb.append( " -runStyle ").append( quote( mode ) );
548         }
549         if ( userAgents != null && !userAgents.trim().isEmpty() )
550         {
551             sb.append( " -userAgents " ).append( quote( userAgents ) );
552         }
553         if ( batch != null && !batch.trim().isEmpty() )
554         {
555             sb.append( " -batch " ).append( quote( batch ) );
556         }
557         // TODO Is addArgumentDeploy(cmd) also needed to get readable test stacktraces with an alternative deploy dir?
558 
559         return sb.toString();
560     }

562     private String quote(String arg) {
563         return StringUtils.quoteAndEscape( arg, '"', new char[] { '"', ' ', '\t', '\r', '\n' } );
564     }

573     /**
574      * @param clazz class to check for classpath resolution
575      * @return The classpath element this class was loaded from
576      */
577     private File getClassPathElementFor( Class<?> clazz )
578     {
579         String classFile = clazz.getName().replace( '.', '/' ) + ".class";
580         ClassLoader cl = Thread.currentThread().getContextClassLoader();
581         if ( cl == null )
582         {
583             cl = getClass().getClassLoader();
584         }
585         URL url = cl.getResource( classFile );
586         getLog().debug( "getClassPathElementFor " + clazz.getName() + " file " + url.toString() );
587         String path = url.toString();
588     
589         if ( path.startsWith( "jar:" ) )
590         {
591             path = path.substring( 4, path.indexOf( "!" ) );
592         }
593         else
594         {
595             path = path.substring( 0, path.length() - classFile.length() );
596         }
597         if ( path.startsWith( "file:" ) )
598         {
599             path = path.substring( 5 );
600             // windauze hack with maven 3 we get those !
601             path = path.replace( "%20", " " );
602         }
603         File file = new File( path );
604         getLog().debug( "getClassPathElementFor " + clazz.getName() + " file " + file.getPath() );
605         return file;
606     }
